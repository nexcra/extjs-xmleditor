 90 /**
 91  * @description Internal method executed every time a text is clicked<BR>
 92  * It check if a node has editabe property, calculates new carret position and places the carret
 93  * @param {Event Object} e Event Object associate with a mouse click.
 94  */
 95 ContentEditor.handleClick = function(e) {
 96 	//TextRange info: http://www.quirksmode.org/dom/range_intro.html
 97 	if (!window.getSelection()) return;
 98 	var node = window.getSelection().anchorNode.parentNode;
 99 	//console.log(node);
100 	//console.log(new XMLSerializer().serializeToString(ContentEditor.current_epanel.views[1].content));
101 	if (!ContentEditor.isNodeOfType(node,0)) return;
102 	
103 	//console.log(ContentEditor.getCaretDetails());
104 
105 	var new_caret_pos = window.getSelection().anchorOffset; //position in the current node
106 	new_caret_pos+=ContentEditor.adjustPos(node,new_caret_pos);
107 	ContentEditor.current_scrollable_component = ContentEditor.current_epanel.getScrollable(node); //need the container for scrolling purposes
108 
109 	// in the case we clicked in a -already-split-node-, after rejoin the position will change accordingly to the previous node
110 	if (ContentEditor.isNodeOfType(node,1)) { //check if it is a -already-split-node-
111 		if (ContentEditor.isNodeOfType(node.prevTextNode,1)) //check if the previous one is as well temporary
112 			new_caret_pos += node.prevTextNode.innerHTML.length; //if yes - adjust the position
113 		node = Ext.get(node).parent().dom; //from now refer to parent
114 		ContentEditor.removeCaret(); //rejoin nodes
115 	}
116 	ContentEditor.displayCaret({node:node,pos:new_caret_pos});
117 };

166 /* defaults to 1 */
167 ContentEditor.nextCharLength = function(node,pos) {
168 	var ret = ContentEditor.isSpecialChar(node.innerHTML,pos);
169 	if (ret>0) return ret;
170 	else return 1;
171 };

173 /* defaults to 1 */	
174 ContentEditor.prevCharLength = function(node,pos) {
175 	if (pos<=0) return 1;
176 	var i=1;
177 	if (node.innerHTML[pos-i]==';') { //might be
178 		while ((pos-i>=0) && (node.innerHTML[pos-i]!='&')) i++; //looking for '&'
179 		if (pos-i<0) return 1;
180 		return i;
181 	}
182 	return 1;
183 };

185 ContentEditor.handleKey = function(e) {
186 	if (!ContentEditor.caret.parentNode) return;
187 	var caret_obj = ContentEditor.getCaretDetails();
188 	if (e.isNavKeyPress()) {
189 		e.preventDefault();
190 		switch (e.getKey()) {
191 			case e.RIGHT: //arrow right 
192 				caret_obj.pos += ContentEditor.nextCharLength(caret_obj.next_node,0);
193 				ContentEditor.displayCaret(caret_obj); 
194 			break;
195 			case e.UP: //arrow up - place it at the end of previous node
196 				caret_obj.pos = -1;
197 				ContentEditor.displayCaret(caret_obj); 
198 			break;
199 			case e.LEFT: //arrow left 
200 				caret_obj.pos -= ContentEditor.prevCharLength(caret_obj.prev_node,caret_obj.pos);
201 				ContentEditor.displayCaret(caret_obj);
202 			break;
203 			case e.DOWN: //arrow down - place it at the end of current node
204 				caret_obj.pos = caret_obj.node.innerHTML.length;
205 				ContentEditor.displayCaret(caret_obj); 
206 			break;
207 			case e.TAB: //arrow down
208 				if (e.shiftKey) {
209 					caret_obj.pos = -1;
210 					ContentEditor.displayCaret(caret_obj); 
211 				} else {
212 					caret_obj.pos = caret_obj.node.innerHTML.length;
213 					ContentEditor.displayCaret(caret_obj); 
214 				}
215 			break;
216 			case e.ENTER:		
217 			break;
218 			default: console.log(e);
219 		}
220 	}
221 	else if (e.isSpecialKey()) {
222 		switch (e.getKey()) {
223 			case e.BACKSPACE: //backspace 
224 				if (caret_obj.pos>0) {  //if there is something to delete - get its length and delete it
225 					var c_width = ContentEditor.prevCharLength(caret_obj.prev_node,caret_obj.prev_node.innerHTML.length);
226 					caret_obj.prev_node.innerHTML = caret_obj.prev_node.innerHTML.substring(0,caret_obj.pos-c_width);
227 					ContentEditor.commitChange(caret_obj.node);
228 				}
229 				e.preventDefault();
230 			break;
231 		}
232 	}
233 	else {
234 		switch (e.getKey()) {
235 
236 			case e.SPACE:
237 				ContentEditor.output(" ");
238 				ContentEditor.commitChange(caret_obj.node);
239 			break;
240 			default:
241 				ContentEditor.output(String.fromCharCode(e.getCharCode()));
242 				ContentEditor.commitChange(caret_obj.node);
243 		}
244 		e.preventDefault();
245 	}
246 	Ext.get(caret_obj.caret).scrollIntoView(ContentEditor.current_scrollable_component); //adjust view	
247 };

252 ContentEditor.isNodeOfType = function(node,type) { //node of type 1 is also node of type 0, vice-versa does not apply!
253 	if (!node) return false;
254 	switch (type) {
255 		case 0: //editable
256 			if (!node.id) return false;
257 			if (node.id.substring(0,ContentEditor.prefix_id.length)==ContentEditor.prefix_id)
258 				return true;
259 		break;
260 		case 1: //editable-temporary
261 			if (!node.id) return false;
262 			var template = ContentEditor.prefix_id + ContentEditor.prefix_tmp_id;
263 			if (node.id.substring(0,template.length)==template)
264 				return true;
265 		break;	
266 		case 2: //caret
267 			if (node.className==ContentEditor.caret_className) return true;
268 		break;
269 	}
270 	return false;
271 };

278 ContentEditor.installHooks = function(node) {
279 	var i;
280 	if (!node) return;
281 	
282 	if (node.nodeType==3 && node.parentNode && node.parentNode.getAttribute(ContentEditor.current_epanel.marker[0])) { //make editable only nodes with marker[0] attribute)
283 		var p_node = node.parentNode; //span
284 		p_node.id = ContentEditor.prefix_id+ContentEditor.counter_id++;
285 		p_node.prevTextNode = ContentEditor.tmp;
286 		p_node.nextTextNode = null;
287 		ContentEditor.tmp = p_node;
288 		if (p_node.prevTextNode)
289 			p_node.prevTextNode.nextTextNode = p_node;
290 	} else if (node.nodeType==3) {
291 		node.parentNode.className = ContentEditor.nEditable_className;
292 	}
293 	
294 	for(i=0; node.childNodes.item(i); i++) 
295 		ContentEditor.installHooks(node.childNodes.item(i));
296 };

299 ContentEditor.removeHooks = function(node) {
300 	var ret = new Array();
301 	if (!node) return ret;
302 	
303 	var i=0;j=0;
304 	
305 	if (ContentEditor.isNodeOfType(node,2)) return ret;
306 	
307 	for (i=node.childNodes.length-1;i>=0;i--) {
308 		var child = node.childNodes.item(i);
309 		var n_child = ContentEditor.removeHooks(child);
310 		for (j=0;j<n_child.length;j++)
311 			node.insertBefore(n_child[j],child);
312 		node.removeChild(child);	
313 	}
314 	
315 	if (ContentEditor.isNodeOfType(node,0)) {
316 		for (j=0;j<node.childNodes.length;j++)
317 			ret[ret.length] = node.childNodes.item(j);
318 		return ret;
319 	}
320 	
321 	ret[0] = node.cloneNode(true);
322 	return ret;
323 };

325 /** 
326 * Returns number that represents number of characters in node that belong to special characters
327 */
328 ContentEditor.adjustPos = function(node, pos) {
329 	var i=0;
330 	var ret = 0;
331 	for (i=0;i<pos;i++) {
332 		var _ret = ContentEditor.isSpecialChar(node.innerHTML,i+ret)
333 		if (_ret) {ret+=_ret; ret--;} //pos does include 1 char per special char so we have to remove it 
334 	}
335 	return ret;
336 };

338 /**
339 * Return false if it is a regular character (valid & character)
340 * Returns numer - if it is a special character; Number = number of characters of the special character
341 */
342 ContentEditor.isSpecialChar = function(str, pos) {
343 	if (pos<0) return 0;
344 	var ret = 0;
345 	if (str[pos]=='&') {
346 		var _j = ret;
347 		while ((str[pos+ret]!=';') && (pos+ret<str.length)) ret++;
348 		ret++;
349 		if (pos+ret>str.length) ret=_j;
350 	} 
351 	return ret;		
352 };

354 /** Splits a node into two parts : (0,l) & (l,+)
355 * return object with references to the two parts
356 * assumption: node.length>1 && 0<l<node.length
357 * the split creates two new nodes within node! Node persists!
358 */
359 ContentEditor.splitTextNode = function(node, l) {
360 	var ret = new Object();
361 	ret.left = document.createElement("span");
362 	ret.left.id = ContentEditor.prefix_id+ContentEditor.prefix_tmp_id+ContentEditor.counter_tmp_id++;
363 	ret.left.innerHTML = node.innerHTML.substring(0,l);
364 	ret.right = document.createElement("span");
365 	ret.right.id = ContentEditor.prefix_id+ContentEditor.prefix_tmp_id+ContentEditor.counter_tmp_id++;
366 	ret.right.innerHTML = node.innerHTML.substring(l);
367 	removeAllChildren(node);
368 	node.appendChild(ret.left);
369 	node.appendChild(ret.right);
370 	
371 	ContentEditor.repairEditableList(node.prevTextNode,ret.left,ret.right);
372 	ContentEditor.repairEditableList(ret.left,ret.right,node.nextTextNode);
373 	return ret;	
374 };

376 /** Rejoins the split done by splitTextNode 
377 * pnode is node that was used in splitTextNode function
378 */
379 ContentEditor.rejoinTextNodes = function(pnode) {
380 	var i=0;
381 	var txt = "";
382 	for (i=0;i<pnode.childNodes.length;i++) {
383 		txt += pnode.childNodes.item(i).innerHTML;
384 		ContentEditor.counter_tmp_id--;
385 	}
386 	removeAllChildren(pnode);
387 	pnode.innerHTML = txt;
388 	
389 	ContentEditor.repairEditableList(pnode.prevTextNode,pnode,pnode.nextTextNode);
390 };

393 /**
394 * Repairs a,b,c nextTextNode/prevTextNode
395 * c can be undefined
396 */	
397 ContentEditor.repairEditableList = function(a,b,c) {
398 	if (a) a.nextTextNode = b;
399 	if ((b) && (c!=undefined)) b.nextTextNode = c;
400 	if (c) c.prevTextNode = b;
401 	if (b) b.prevTextNode = a;
402 };

404 /**
405 * Method return an object that represents caret position within given dom
406 */
407 ContentEditor.getCaretDetails = function(node) {
408 	var caret_ptr = null;
409 	if (node) caret_ptr = Ext.get(node).child("span."+ContentEditor.caret_className);
410 	else caret_ptr = Ext.get(ContentEditor.caret);
411 	
412 	if (caret_ptr)
413 		return {
414 			caret : caret_ptr.dom,
415 			node : caret_ptr.dom.parentNode,
416 			prev_node : caret_ptr.dom.previousSibling,
417 			pos : caret_ptr.dom.previousSibling!=null?caret_ptr.dom.previousSibling.innerHTML.length:null,
418 			next_node : caret_ptr.dom.nextSibling,
419 		};
420 
421 	return null;
422 };

424 ContentEditor.removeCaret = function() {
425 	if(ContentEditor.caret.parentNode != null) {
426 		// Remove from current parent
427 		var parent = ContentEditor.caret.parentNode;
428 		ContentEditor.caret.parentNode.removeChild(this.caret);
429 		ContentEditor.rejoinTextNodes(parent);
430 	}
431 };

433 /**
434 * Displayes caret in a given node
435 */
436 ContentEditor.displayCaret = function(caret_obj) {
437 	if (!caret_obj) return;
438 	var node = caret_obj.node;
439 	var pos = caret_obj.pos;
440 	ContentEditor.removeCaret();
441 	if (pos<0) {
442 		if (node.prevTextNode) {
443 			node = node.prevTextNode;
444 			pos = node.innerHTML.length;
445 		} else pos=0;
446 	}
447 	if (pos>node.innerHTML.length) {
448 		if (node.nextTextNode) {
449 			node = node.nextTextNode;
450 			pos = 0;
451 		} else pos = node.innerHTML.length;
452 	}		
453 	var tmp = ContentEditor.splitTextNode(node,pos);
454 	node.insertBefore(ContentEditor.caret,tmp.right);
455 	ContentEditor.positionChanges(node,pos,ContentEditor.createIdentifier(ContentEditor.caret));
456 };

492 Array.prototype.cut = function(from, to) {
493   var rest = this.slice((to || from) + 1 || this.length);
494   this.length = from < 0 ? this.length + from : from;
495   return this.push.apply(this, rest);
496 };

501 Array.prototype.remove = function(elem) {
502 	var i=0;
503 	while (i<this.length && this[i]!=elem) i++;
504 	return this.cut(i);
