1 /**
  2 * After processing each final TextNode is wrapped with SPAN 
  3 */
  4 
  5 var XMLPanelEditable = Ext.extend(PanelEditable, {
  6 	url : null,
  7 	rdom : null, //resulting dom (after applying all transformations and transforming the controller document)
  8 	xslt : null, //source xslt
  9 	init_arr : null, //transformations to apply to xslt
 10 	rxslt : null, //resulting xslt
 11 	hashTable: null, //maps: id->node_ref
 12 	controller : null,
 13 	initComponent: function () {
 14 		XMLPanelEditable.superclass.initComponent.apply(this, arguments);
 15 	},
 16 	
 17 	/**
 18 	* Loads XSLT from an URL to be used for this view 
 19 	*/
 20 	loadXSLTurl: function(url) {
 21 		this.url = url;
 22 		this.init_arr = arguments;
 23 		if (!url) {
 24 			this.loadXSLTdom(null);
 25 			return;
 26 		}
 27 		var that = this;
 28 		loadXML(url, function(dom) {that.loadXSLTdom(dom);} );
 29 	},
 30 	
 31 	/**
 32 	* Loads XSLT from dom to be used for this view 
 33 	*/
 34 	loadXSLTdom: function(dom) {
 35 		if (arguments.length>1)
 36 			this.init_arr = arguments;
 37 		this.rxslt = null;
 38 		this.xslt = dom;
 39 		this.process();
 40 	},
 41 	
 42 	/**
 43 	* Pulls markers from controller
 44 	*/
 45 	pullMarkers: function() {
 46 		this.marker = this.controller.marker;
 47 		this.markerClassName = this.controller.markerClassName;
 48 	},
 49 	
 50 	refreshContent: function() { //if contains caret it will be vanished as the method generates a totally new dom tree
 51 		this.unhook(this.content);
 52 		removeAllChildren(this.content); //removes caret as well!
 53 		this.process();
 54 	},
 55 	
 56 	addSnapshot: function() {
 57 		XMLPanelEditable.superclass.addSnapshot.apply(this, arguments);
 58 	},
 59 	
 60 	recallSnapshot: function(step) {
 61 		XMLPanelEditable.superclass.recallSnapshot.apply(this, arguments);
 62 		this.rdom = this.content;
 63 		this.createHashTable(this.content);
 64 	},
 65 	
 66 	/**
 67 	* function to be called internally by XMLEditor
 68 	* Note: arguments[0] is a new value for a node described by markers and it does not contain caret!
 69 	*/
 70 	updateElement: function(/*value + markers*/) {
 71 		var caret = ContentEditor.getCaretDetails(this.content); //will be null in all cases except of current content having caret
 72 		
 73 		if (arguments[2]) {
 74 			if (!this.hashTable[arguments[1]+arguments[2]+arguments[3]]) return;
 75 			this.hashTable[arguments[1]+arguments[2]+arguments[3]].innerHTML = arguments[0]; //xml context
 76 		}else {
 77 			if (!this.hashTable[arguments[1]]) return;
 78 			this.hashTable[arguments[1]].innerHTML = arguments[0]; //xml context
 79 		}
 80 		
 81 		ContentEditor.displayCaret(caret);
 82 	},
 83 	
 84 	parseDocument: function(root) {
 85 		if (!root) return;
 86 		
 87 		if (root.nodeType==3) { //for each textNode wrap it into span!
 88 			var bc = XMLAux.parseElement(root.previousSibling);
 89 			var x = document.createElement('span');
 90 			x.appendChild(root.cloneNode(true));
 91 			root.parentNode.replaceChild(x,root);
 92 			//if the textnode is the follower of our breadcrumb - add attribute
 93 			if (bc) {
 94 				x.setAttribute(this.marker[0],bc.id);
 95 				x.className = this.markerClassName[0];
 96 				if (bc.attr) {
 97 					x.setAttribute(this.marker[1],bc.attr);
 98 					x.setAttribute(this.marker[2],bc.attr_ns);
 99 					x.className = this.markerClassName[1];
100 				}
101 			}
102 		}
103 	
104 		var c = root.childNodes;
105 		for (var i=0;i<c.length;i++)
106 			this.parseDocument(c.item(i));
107 	},
108 	
109 	createHashTable: function(root) {
110 		this.hashTable = new Array();
111 		this._createHashTable(root);
112 	},
113 	
114 	/**
115 	* Hash table has to be created once all nodes are within document (references are valid!)
116 	*/
117 	_createHashTable: function(root) {
118 		if (!root) return;
119 	
120 		if (root.nodeType==3) { //for each textNode wraped it into span!
121 			var pnode = root.parentNode;
122 			var id = pnode.getAttribute(this.marker[0]);
123 			var attr = pnode.getAttribute(this.marker[1]);
124 			var attr_ns = pnode.getAttribute(this.marker[2]);
125 			if (!attr) this.hashTable[id] = pnode;
126 			else this.hashTable[id+attr+attr_ns] = pnode;
127 		}
128 
129 		var c = root.childNodes;
130 		for (var i=0;i<c.length;i++)
131 			this._createHashTable(c.item(i));
132 	},
133 	
134 	/**
135 	* Process & Display children
136 	*/
137 	process: function() {
138 		if (!this.controller) {
139 			console.log("Error: XMLPanelEditable not attached to XMLEditor!");
140 			return;
141 		}
142 		this.pullMarkers();
143 
144 		var doc = this.controller.getDom();
145 
146 		if (!this.rxslt && this.xslt) { //create resulting xslt TODO
147 			this.rxslt = this.xslt;
148 			for (var i=1;this.init_arr && i<this.init_arr.length;i++) {
149 				var processor = new XSLTProcessor(); //create XSLTProcessor
150 				processor.importStylesheet(this.init_arr[i]);
151 				this.rxslt = processor.transformToDocument(this.rxslt);
152 			}
153 			
154 			/** And at the end always apply XSLTTransform! **/
155 			var processor = new XSLTProcessor(); //create XSLTProcessor
156 			processor.importStylesheet(XSLTSet.XSLTTransform);
157 			this.rxslt = processor.transformToDocument(this.rxslt);
158 		}
159 		if (this.rxslt) { //create resulting document rxslt(doc)
160 			var processor = new XSLTProcessor();
161 			processor.importStylesheet(this.rxslt);
162 			this.rdom = processor.transformToDocument(doc);
163 		} else {
164 			console.log("Error: no transformation!");
165 			return;
166 		}
167 		this.parseDocument(this.rdom); //get id's properly //have to construct hash table after inserted!
168 		this.initContent(new XMLSerializer().serializeToString(this.rdom));	
169 		this.rdom = this.content;
170 		this.createHashTable(this.rdom);
171 		this.makeEditable();
172 	},
173 });
174 
175 Ext.reg('XMLPanelEditable', XMLPanelEditable);
