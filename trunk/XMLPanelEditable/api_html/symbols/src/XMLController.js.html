1 /* class definition */
  2 /* Keeps original XML file (transformed using XMLTransformation) */
  3 
  4 var XMLController = function() {
  5 	this.init();
  6 };
  7 
  8 Ext.extend(XMLController, Ext.Element, {
  9 	//Variables
 10 	views : null,
 11 	init_arr : null,
 12 	textNodeHashTable : null, //keeps a relation id->dom
 13 	nodeHashTable : null,
 14 	snapshot : null,	
 15 	marker : ['fxe_id','fxe_attr','fxe_attr_ns'], //the first marker makes a node being editable, the following are optional in order to differ nodes
 16 	markerClassName : ['fxe_editable','fxe_editable_a','fxe_editable_a'],
 17 	
 18 	//Functions
 19 	init: function() {
 20 		this.views = new Array();
 21 	},
 22 	
 23 	getDom: function() {
 24 		return this.dom;
 25 	},
 26 	
 27 	/**
 28 	* Loads XML from an URL to be used for the base
 29 	*/	
 30 	loadXMLurl: function(url) {
 31 		this.init_arr = arguments;
 32 		var that = this;
 33 		loadXML(url, function(dom) {that.loadXMLdom(dom);} );
 34 	},
 35 
 36 	/**
 37 	* Loads XML from a dom to be used for the base
 38 	*/		
 39 	loadXMLdom: function(dom) {
 40 		if (arguments.length>1)
 41 			this.init_arr = arguments;	
 42 		this.dom = dom;
 43 		this.process();
 44 	},
 45 	
 46 	/**
 47 	* Adds a new view for the current XML
 48 	*/
 49 	addView: function(view) {
 50 		this.views[this.views.length] = view;
 51 		view.controller = this;
 52 		//link IDs with breadcrumbs
 53 	},
 54 
 55 	/**
 56 	* Removes a view for the current XML
 57 	*/
 58 	removeView: function(view) {
 59 		this.views = this.views.remove(view);
 60 	},
 61 	
 62 	initSnapshot: function() {
 63 		this.snapshot = new Array();
 64 		this.snapshot.current_snap_pos = undefined;
 65 	},
 66 	
 67 	addSnapshot: function() {
 68 		if (this.snapshot.current_snap_pos!=undefined) {
 69 			this.snapshot.trim(this.snapshot.current_snap_pos+1);
 70 		}
 71 
 72 		var pos = this.snapshot.length;
 73 		this.snapshot[pos] = new Object();
 74 		var snap = this.dom.cloneNode(true);
 75 		this.snapshot[pos].snap = snap;
 76 		this.snapshot.current_snap_pos = pos;
 77 		
 78 		for (var i=0;i<this.views.length;i++)
 79 			this.views[i].addSnapshot.apply(this.views[i],arguments);
 80 	},
 81 	
 82 	recallSnapshot: function(step) {
 83 		var s_n = this.snapshot.current_snap_pos+step;
 84 		if (s_n>=this.snapshot.length) return;
 85 		if (s_n<0) return;
 86 		var ptr = this.snapshot[s_n];
 87 		this.snapshot.current_snap_pos = s_n;
 88 
 89 		removeAllChildren(this.dom);
 90 
 91 		this.dom = ptr.snap.cloneNode(true);
 92 		this.parseDocument(this.dom);
 93 		
 94 		for (var i=0;i<this.views.length;i++)
 95 			this.views[i].recallSnapshot.apply(this.views[i],arguments);
 96 			
 97 		if (s_n==0) ContentEditor.initCaret(); //the very first snapshot does not have caret so once we are at the very first snapshot we have to re-initiate the caret as we destroyed it by issuing removeAllChildren
 98 	},
 99 	
100 	/**
101 	* Updates value of an element and propagates to the rest of views
102 	* view - view that the issue originates from
103 	*/
104 	updateElement: function(/*value + markers values*/) {
105 		//console.log(arguments);
106 		if (arguments[2]) { //looking for node (marker[1])
107 			this.nodeHashTable[arguments[1]].setAttributeNS(arguments[3],arguments[2],arguments[0]);
108 		}
109 		else { //looking for textNode
110 			this.textNodeHashTable[arguments[1]].nodeValue = arguments[0];
111 		}
112 
113 		for (var i=0;i<this.views.length;i++)
114 			this.views[i].updateElement.apply(this.views[i],arguments);
115 		
116 		this.addSnapshot();
117 	},
118 	
119 	parseDocument: function(root) {
120 		this.nodeHashTable = new Array();
121 		this.textNodeHashTable = new Array();
122 		this._parseDocument(root);
123 	},
124 	
125 	_parseDocument: function(root) {
126 		if (!root) return;
127 		
128 		var bc = XMLAux.parseElement(root);
129 		
130 		if (bc && root.nodeType==7)
131 			this.textNodeHashTable[bc.id] = root.nextSibling;
132 
133 		else if (bc && root.nodeType==1) this.nodeHashTable[bc.id] = root;
134 		
135 		var c = root.childNodes;
136 		for (var i=0;i<c.length;i++)
137 			this._parseDocument(c.item(i));
138 	},
139 	
140 	updateViews: function() {
141 		this.refreshContent();
142 	},	
143 	
144 	refreshContent: function() {
145 		for (var i=0;i<this.views.length;i++)
146 			this.views[i].refreshContent.apply(this.views[i]);
147 		ContentEditor.initCaret();
148 	},
149 	
150 	getScrollable: function(node) {
151 		return Ext.get(node).up("div.x-panel-body");
152 	},
153 	
154 	/**
155 	* Function transforms the dom using removeBC transformation (reverse of xml_transformation)
156 	* Returns: dom!
157 	*/
158 	output: function() {
159 		var processor = XSLTProcessor(); //create XSLTProcessor
160 		processor.importStylesheet(XSLTSet.rXMLTransform);
161 		var ret = processor.transformToDocument(this.dom);
162 		//return ret
163 		console.log(new XMLSerializer().serializeToString(ret));
164 	},
165 	
166 	process: function() {
167 		//append IDs
168 		
169 		for (var i=1;this.init_arr && i<this.init_arr.length;i++) {
170 			var processor = new XSLTProcessor(); //create XSLTProcessor
171 			processor.importStylesheet(this.init_arr[i]);
172 			this.dom = processor.transformToDocument(this.dom);
173 		}	
174 		/** And at the end always apply XMLTransform! */
175 		var processor = new XSLTProcessor(); //create XSLTProcessor
176 		processor.importStylesheet(XSLTSet.XMLTransform);
177 		this.dom = processor.transformToDocument(this.dom);
178 
179 		this.parseDocument(this.dom);
180 		
181 		this.initSnapshot();
182 		this.addSnapshot();
183 	},
184 	
185 });
