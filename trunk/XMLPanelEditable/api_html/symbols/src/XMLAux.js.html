  6 function loadXML(url,callback) {
  7 	var req = new XMLHttpRequest();  
  8 	req.open('GET', url, false);   
  9 	req.send(null);  
 10 	if(req.status == 0) {
 11 		var _v = (new DOMParser()).parseFromString(req.responseText, "text/xml");
 12 		callback(_v);
 13 	}
 14 }

 16 /* dynamic JavaScript loader */
 17 function jsLoader(url) {
 18 	var scriptObj = document.createElement("script");
 19 	scriptObj.setAttribute("src", url /*+ '&noCacheIE=' + (new Date()).getTime()*/);
 20 	scriptObj.setAttribute("type", "text/javascript");
 21 	document.getElementsByTagName("head").item(0).appendChild(scriptObj);
 22 }

 37 function XMLAux() {};
 38 
 39 XMLAux.bc_text_start = "fxTextStart";
 40 XMLAux.bc_text_end = "fxTextEnd";
 41 XMLAux.bc_attr_start = "fxAttrStart";
 42 XMLAux.bc_attr_end = "fxAttrEnd";
 43 XMLAux.editor_id = "id"; //node id appended by XMLTransform
 44 XMLAux.editor_URI = "factonomy.com/schema/xmleditor"; //editor namespacce
 45 
 46 
 47 XMLAux.parseElement = function(node) {
 48 	if (!node) return null;
 49 	//console.log(node.nodeName);
 50 	var ret = new Object();
 51 	if (node.nodeType==7 && (node.target==XMLAux.bc_text_start)) {
 52 		var id_start_pos = node.data.indexOf("id=")+3;
 53 		ret.id = node.data.substring(id_start_pos);
 54 	}
 55 	else if (node.nodeType==7 && (node.target==XMLAux.bc_attr_start)) {
 56 		var id_start_pos = node.data.indexOf("id=")+3;
 57 		var id_end_pos = node.data.indexOf(" attr=");
 58 		var attr_start_pos = node.data.indexOf(" attr=")+6;
 59 		var attr_end_pos = node.data.indexOf(" attr_ns=");
 60 		var attr_ns_start_pos = node.data.indexOf(" attr_ns=")+9;
 61 		ret.id = node.data.substring(id_start_pos,id_end_pos);
 62 		if (attr_start_pos>0) {
 63 			ret.attr = node.data.substring(attr_start_pos,attr_end_pos);
 64 			ret.attr_ns = node.data.substring(attr_ns_start_pos);
 65 		}
 66 	}
 67 	else if (node.nodeType==1)
 68 		ret.id = node.getAttributeNS(XMLAux.editor_URI,XMLAux.editor_id);
 69 	
 70 	ret.name = node.target;
 71 	
 72 	if (ret.id) return ret;
 73 	return null;
 74 };
